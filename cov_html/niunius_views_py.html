192 def post(self, request, slug): 

193 """ 

194 If no shopping cart in the session, create and save one. 

195 Then add to the cart the product with the chosen quantity. 

196 

197 If a shopping cart already saved in the session, 

198 just add the product with the chosen quantity. 

199 

200 If the product already added to the cart, just update its quantity. 

201 """ 

202 product = Product.objects.get(slug=slug) 

203 qty = int(request.POST.get("qty")) 

204 if "cart" not in request.session: 

205 cart = ShoppingCart.objects.create() 

206 request.session["cart"] = cart.id 

207 CartItem.objects.create(product=product, quantity=qty, cart=cart) 

208 return redirect("cart") 

209 else: 

210 cart = ShoppingCart.objects.get(pk=request.session["cart"]) 

211 try: 

212 item = cart.cartitem_set.get(product_id=product.pk) 

213 except CartItem.DoesNotExist: 

214 CartItem.objects.create(product=product, quantity=qty, cart=cart) 

215 return redirect("cart") 

216 item.quantity += qty 

217 item.save() 

218 cart.cartitem_set.get(product_id=product.pk).quantity += qty 

219 cart.cartitem_set.get(product_id=product.pk).save() 

220 return redirect("cart")

231class ShoppingCartView(View):

233 """ 

234 Get items from the shopping cart saved in the session.

236 

237 If no shopping cart saved, display the message about empty cart. 

238 """ 

239 if request.session.get("cart"): 

240 cart = ShoppingCart.objects.get(pk=request.session.get("cart")) 

241 items = cart.cartitem_set.all().order_by("pk") 

242 total = cart.order_total() 

243 ctx = {"items": items, "total": total} 

244 return render(request, "niunius/cart.html", ctx) 

245 else: 

246 return render(request, "niunius/cart.html")

248 def post(self, request): 

249 """ 

250 Update quantity for given item 

251 and re-calculate the total value of all items in the cart. 

252 """ 

253 cart = ShoppingCart.objects.get(pk=request.session.get("cart")) 

254 items = cart.cartitem_set.all().order_by("pk") 

255 qty = int(request.POST.get("qty"))

257 item = cart.cartitem_set.get(product=product) 

258 item.quantity = qty 

259 item.save() 

260 total = cart.order_total() 

261 ctx = {"items": items, "total": total} 

262 return render(request, "niunius/cart.html", ctx)

334 def post(self, request): 

335 """ 

336 Add to the database new Order object. 

337 If added successfully, redirect to the confirmation page. 

338 Otherwise, return the page with an empty form. 

339 

340 After the order is created, delete the assigned cart from the session 

341 and decrease the stock field for ordered products. 

342 """ 

343 form = OrderForm(request.POST) 

344 buyer_form = BuyerForm(request.POST) 

345 if form.is_valid() and buyer_form.is_valid(): 

346 address_city = form.cleaned_data["address_city"] 

347 address_zipcode = form.cleaned_data["address_zipcode"] 

348 address_street = form.cleaned_data["address_street"] 

349 address_country = form.cleaned_data["address_country"] 

350 cart = ShoppingCart.objects.get(pk=request.session.get("cart")) 

351 order = Order.objects.create( 

352 cart=cart, 

353 buyer=request.user, 

354 address_city=address_city, 

355 address_zipcode=address_zipcode, 

356 address_street=address_street, 

357 address_country=address_country, 

358 ) 

359 buyer = request.user 

360 buyer.first_name = buyer_form.cleaned_data["first_name"] 

361 buyer.last_name = buyer_form.cleaned_data["last_name"] 

362 buyer.email = buyer_form.cleaned_data["email"] 

363 buyer.save() 

364 for item in cart.cartitem_set.all(): 

365 item.product.stock -= item.quantity 

366 item.product.save() 

367 del request.session["cart"] 

368 return redirect("confirmation", order.pk) 

369 return render( 

370 request, "niunius/order_form.html", {"form": form, "buyer_form": buyer_form} 

371 )
